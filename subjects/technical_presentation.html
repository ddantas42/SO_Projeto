fork() & Memória Compartilhada

Como o IPC fica acessível aos filhos

create_process()

pid_t create_process(...) {

pid = fork();

if (pid < 0) {

perror("fork"); exit(1);

}

if (pid == 0) { // CHILD

register_sig(SIGINT, SIG_IGN);

register_sig(SIGTSTP, SIG_IGN);

thread_creator(board, fn, N);

exit(SUCCESS);

}

// PARENT retorna pid

return pid;

}

initialize_sharedboard()

// Criar segmento SHM

shmid = shmget(IPC_PRIVATE,

sizeof(t_sharedboard),

IPC_CREAT | 0666);

// Attached ao espaço do processo

*board = shmat(shmid, NULL, 0);

// Inicializar mutex

board->board_mutex =

PTHREAD_MUTEX_INITIALIZER;

// Filhos herdam ptr!

🎯 Insight Crítico

fork() copia o espaço de endereços do PAI. O pointer board aponta para o MESMO segmento de SHM em ambos os processos. Isto é o truque de IPC!

Signal Handling com sigaction()

Porque SIGINT e SIGTSTP funcionam globalmente

register_sig() — Setup

void register_sig(int signum,

void (*handler)(int)) {

struct sigaction sa;

sa.sa_handler = handler;

sigemptyset(&sa.sa_mask);

sa.sa_flags = 0;

sigaction(signum, &sa, NULL);

}

sig_handler() — Ação

void sig_handler(int signum) {

if (signum == SIGINT) {

if (board != NULL)

board->stop_signal = 1;

}

if (signum == SIGTSTP) {

if (board != NULL)

board->pause_analysis =

!board->pause_analysis;

}

}

Main Process (PAI)

  • ✓ Registra sig_handler para SIGINT
  • ✓ Registra sig_handler para SIGTSTP
  • ✓ Modifica board→stop_signal ou pause_analysis
  • ✓ Threads leem estes flags

Child Processes (FILHOS)

  • ✓ Registram SIG_IGN para SIGINT
  • ✓ Registram SIG_IGN para SIGTSTP
  • ✗ Ignoram sinais! Não reagem
  • ⚡ Ainda usam board (shared memory)

thread_creator() — Multithreading

VLA (Variable Length Array) e t_thread_args

void thread_creator(t_sharedboard *board,

void *(*thread_func)(void *),

int number_of_threads) {

// VLA — declarado na stack!

pthread_t tid[number_of_threads];

t_thread_args args[number_of_threads];

srand(time(NULL) / getpid());

// Criar threads

for (int i = 0; i < number_of_threads; i++) {

args[i].board = board; // Shared!

args[i].thread_id = i;

pthread_create(&tid[i], NULL,

thread_func, (void *)&args[i]);

}

// Aguardar todas as threads

for (int i = 0; i < number_of_threads; i++) {

pthread_join(tid[i], NULL);

}

}

⚠️ VLA Stack

Arrays na stack! Precisa C99+. Não é portável mas funciona aqui.

🎯 t_thread_args

Cada thread recebe o seu ID + pointer para shared board

✓ pthread_join()

Aguarda que TODAS as threads terminem antes de sair do processo

exploration_thread() — Drone Loop

Produtor: gera, verifica, deposita

void *exploration_thread(void *arg) {

t_thread_args *args = arg;

t_sharedboard *board = args->board;

t_sample sample;

int id = args->thread_id;

// Loop principal — continua enquanto stop_signal == 0

while (!board->stop_signal) {

// 1. Simular coleta (5 segundos)

sleep(DRONE_FINDING_TIME);

// 2. Gerar amostra com dados aleatórios

sample = generate_sample(id);

pthread_mutex_lock(&board->board_mutex);

// 3. Verificar se há espaço

if (board->count >= STORAGE_CAPACITY) {

pthread_mutex_unlock(&board->board_mutex);

logger(DRONE_LOG, id,

"Waiting for space...\\n");

continue; // retry no próximo loop

}

// 4. Depositar amostra

deposit_sample(board, &sample);

pthread_mutex_unlock(&board->board_mutex);

logger(DRONE_LOG, id,

"Deposited sample\\n");

}

logger(DRONE_LOG, id, "Stopping..\\n");

return NULL;

}

analysis_thread() — Consumidor Loop

Consumidor: retira, analisa, grava CSV

void *analysis_thread(void *arg) {

t_thread_args *args = arg;

t_sharedboard *board = args->board;

t_sample sample;

int id = args->thread_id;

while (!board->stop_signal) {

pthread_mutex_lock(&board->board_mutex);

// 1. Verificar se há amostras para analisar

if (board->count <= 0) {

pthread_mutex_unlock(&board->board_mutex);

logger(ANALYSIS_LOG, id,

"Waiting for samples...\\n");

sleep(1); // Reduz busy-wait

continue;

}

// 2. Retirar amostra

sample = get_sample(board);

pthread_mutex_unlock(&board->board_mutex);

// 3. Analisar (fora da seção crítica!)

analyse_sample(&sample, id);

// 4. Gravar CSV

append_sample_csv(&sample);

// 5. Verificar pausa (Ctrl+Z)

while (board->pause_analysis &&

!board->stop_signal) {

logger(ANALYSIS_LOG, id,

"Analysis paused...\\n");

sleep(1);

}

}

logger(ANALYSIS_LOG, id, "Stopping..\\n");

return NULL;

}

deposit_sample() & get_sample()

Circular buffer com in/out/count

deposit_sample() — Produtor

void deposit_sample(t_sharedboard *b,

t_sample *sample) {

// Timestamp da deposição

clock_gettime(CLOCK_ID,

&sample->deposited_time);

// Inserir na posição 'in'

b->samples[b->in] = *sample;

// Avanço circular

b->in = (b->in + 1)

% STORAGE_CAPACITY;

b->count++;

}

get_sample() — Consumidor

t_sample get_sample(

t_sharedboard *b) {

// Copiar da posição 'out'

t_sample *ptr =

&b->samples[b->out];

t_sample local = *ptr;

// Limpar slot

init_sample(ptr);

// Avanço circular

b->out = (b->out + 1)

% STORAGE_CAPACITY;

b->count--;

return local;

}

Tricky Cases & Edge Cases

O que pode dar errado e porquê

❌ fflush(NULL) Omitido

Output duplicado após fork() porque buffers não foram esvaziados

❌ Não Unlock Antes de sleep()

Drone com mutex locked e sleep(1) bloqueia todas as análises!

⚠️ EINTR em waitpid()

Sinais interrompem waitpid(). Precisa retry em loop se errno==EINTR

⚠️ Buffer Cheio vs Vazio

count >= STORAGE: drones esperam | count <= 0: análise espera

✓ Volatile sig_atomic_t

stop_signal e pause_analysis são volatile — flags atómicas, sem race conditions

✓ SHM Persiste

Shared memory SEM shmctl() IPC_RMID deixa orphan — precisa limpeza

Cleanup & Lições Aprendidas

free_sharedboard() e takeaways

free_sharedboard() — Cleanup

void free_sharedboard(t_sharedboard *board) {

if (board == NULL) return;

// Destruir mutexes

pthread_mutex_destroy(&board->board_mutex);

pthread_mutex_destroy(&board->log_mutex);

// Desattach da SHM

shmdt(board);

// Nota: shmctl(id, IPC_RMID) não chamado aqui!

}

🎯 Key Takeaways

  • ✓ fork() copia espaço — SHM é herdado
  • ✓ sigaction() é melhor que signal()
  • ✓ fflush antes de fork evita duplicação
  • ✓ Mutex protege seções críticas
  • ✓ Unlock ANTES de sleep/wait
  • ✓ volatile sig_atomic_t para flags

⚡ Design Patterns

  • ✓ Producer-Consumer com circular buffer
  • ✓ Mutex para seções críticas apenas
  • ✓ sleep() reduz CPU busy-wait
  • ✓ Check-then-act com mutex protegido
  • ✓ IPC via shared memory + semáforos
  • ✓ Signal handlers para controlo global
